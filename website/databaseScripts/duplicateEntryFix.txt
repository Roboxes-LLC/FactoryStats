# Detect duplicate entries
SELECT stationId, shiftId, dateTime, count, COUNT(*) AS cnt
FROM count
GROUP BY stationId, shiftId, dateTime
HAVING cnt > 1;

# Delete duplicate entries
# Note: Make count key (stationId-shiftId-dateTime) UNIQUE to permanently fix.
DELETE t1
FROM count t1
JOIN count t2
  ON t1.stationId = t2.stationId
 AND t1.shiftId = t2.shiftId
 AND t1.dateTime = t2.dateTime
 AND t1.entryId > t2.entryId;
